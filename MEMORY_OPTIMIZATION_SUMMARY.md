def _cleanup_memory(self):
    """Enhanced memory cleanup with cooldown"""
    try:
        current_time = time.time()
        # Only cleanup if enough time has passed
        if current_time - self.last_cleanup_time < self.cleanup_cooldown:
            return

        gc.collect()
        import torch
